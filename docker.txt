build docker container
docker build -t node-app-image .

list all docker volumes
docker volume ls

list all running docker containers
docker ps

list all docker containers
docker ps -a

delete docker container
docker rm containerName

delete volume
docker volume rm volumeName

delete all non running volumes
docker volume prune

run docker container
${pwd} for current folder address
:ro for makeing our container Readonly if not yhen any chabge in dockercontainer will be reflected in our codebase
--env will set enviroment variable inside our container
-d for detach mode
docker run -v ${pwd}:/app:ro -v /app/node_modules --env ./.env -p 3000:3000 -d --name node-app  node-app-image
